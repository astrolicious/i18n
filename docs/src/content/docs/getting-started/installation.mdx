import { Tabs, TabItem } from '@astrojs/starlight/components';

:::caution[Prerequisites]
- `astro` must be at least on version `4.14.0` or `5.0.0`
- `tsconfig.json` `compilerOptions.strict` should be set to `true`. If you're using Astro tsconfig presets, use either `astro/tsconfigs/strict` or `astro/tsconfigs/strictest`
:::

## Automatic installation

Run the following command in your terminal using your favorite package manager.

<Tabs>
    <TabItem label="pnpm" icon="pnpm">
    ```bash
    pnpm astro add @astrolicious/i18n
    ```
    </TabItem>
    <TabItem label="npm" icon="seti:npm">
    ```bash
    npm run astro add @astrolicious/i18n
    ```
    </TabItem>
    <TabItem label="yarn" icon="seti:yarn">
    ```bash
    yarn astro add @astrolicious/i18n
    ```
    </TabItem>
</Tabs>

This will update your `astro.config.mjs` with the following content:

```js title="astro.config.mjs" ins={2,6}
import { defineConfig } from "astro/config"
import i18n from "@astrolicious/i18n"

export default defineConfig({
    integrations: [
        i18n()
    ]
})
```

## Manual installation

Install the necessary dependencies:

<Tabs>
    <TabItem label="pnpm" icon="pnpm">
    ```bash
    pnpm add @astrolicious/i18n i18next
    ```
    </TabItem>
    <TabItem label="npm" icon="seti:npm">
    ```bash
    npm install @astrolicious/i18n i18next
    ```
    </TabItem>
    <TabItem label="yarn" icon="seti:yarn">
    ```bash
    yarn add @astrolicious/i18n i18next
    ```
    </TabItem>
</Tabs>

Update your `astro.config.mjs` with the following content:

```js title="astro.config.mjs" ins={2,6}
import { defineConfig } from "astro/config"
import i18n from "@astrolicious/i18n"

export default defineConfig({
    integrations: [
        i18n()
    ]
})
```
